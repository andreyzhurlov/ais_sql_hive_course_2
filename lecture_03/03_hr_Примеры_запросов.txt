--Найти сотрудников, у которых длина фамилии больше 8 символов.
SELECT 
    employee_id,
    last_name,
    LENGTH(last_name) AS last_name_length
FROM 
    hr.employees
WHERE 
    LENGTH(last_name) > 8;

select
	region_id
  , region_name
  , case 
        when region_name = 'Asia' THEN 'Азия'
	    else 'Не Азия'
	end
  , case region_name
        when 'Asia' then 'Азия'
		else 'Не Азия'
	end
from hr.regions;

select
    first_name
  , last_name
  , salary
  , case
        when salary between 0.00 and 4999.99
            then 'Low salary'
	    when salary between 5000.00 and 9999.99
	        then 'Normal salary'
		when salary between 10000.00 and 14999.99
	        then 'High salary'
		else 'TOP'
	end
from hr.employees
order by employee_id
limit 10;

select
    emp.employee_id
  , emp.first_name
  , emp.last_name
  , emp.salary
  , emp.job_id
  , jobs.job_title
from hr.employees as emp
left join jobs
on emp.job_id = jobs.job_id and
   emp.salary between jobs.min_salary and jobs.max_salary;
